# Backjoon 2742

while 문

```java

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int i = 1;
        
        while (i <= n) {
            System.out.println(n);
            n--;
        }
        sc.close();  
    }
}

```

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        for (int i = n; i >=1; i--) {
            System.out.println(i);
        }
        sc.close();
